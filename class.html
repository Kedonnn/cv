class Rectangle { constructor (height, width) { this.height = height; this.width = width; } } //constructor helps us to initialize the class const myRectangle = new Rectangle (3,5) const 1Rectangle = new Rectangle (4,2) const 2Rectangle = new Rectangle (6,5) const 3Rectangle = new Rectangle (12,15) const 4Rectangle = new Rectangle (7,11) console.log("4th Rectangle's width is ", 4Rectangle.width) // the "class" or Object oriented programming can encapsulate your code // the "class" or Object oriented programming can use inheritance in your code
